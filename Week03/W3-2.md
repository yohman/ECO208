Add color

library(ggplot2)

# create a scatter plot

###

Stretch the points
```r
# import ggplot
library(ggplot2)

# create a scatter plot
ggplot(data = gapminder, aes(x = gdpPercap, 
                             y = lifeExp, 
                             color=continent,
)) +
  geom_point()  + scale_x_log10()

```

###

Make the points semi-transparent
```r
# import ggplot
library(ggplot2)

# create a scatter plot
ggplot(data = gapminder, aes(x = gdpPercap, 
                             y = lifeExp, 
                             color=continent,
)) +
  geom_point(alpha = 0.5)  + scale_x_log10()

```
